<?xml version="1.0" encoding="UTF-8"?>
<configuration>
    <system.webServer>
        <rewrite>
            <rewriteMaps>
                <rewriteMap name="MapProtocol">
                    <add key="on" value="https" />
                    <add key="off" value="http" />
                </rewriteMap>
            </rewriteMaps>
            <rules>
                <clear />
                <rule name="ScoreSharpYARPProxy" stopProcessing="true">
                    <match url="api/(.*)" />
                    <action type="Rewrite" url="https://172.28.251.76/scoresharp_api/{R:1}" />
                    <serverVariables>
                        <set name="HTTP_X_FORWARDED_PROTO" value="{MapProtocol:{HTTPS}}" />
       			<set name="HTTP_X_FORWARDED_PREFIX" value="/scoresharp_yarp/api" />
			<set name="HTTP_X_FORWARDED_HOST" value="{HTTP_HOST}" />
                    </serverVariables>
                </rule>
            </rules>
        </rewrite>
    </system.webServer>
</configuration>
